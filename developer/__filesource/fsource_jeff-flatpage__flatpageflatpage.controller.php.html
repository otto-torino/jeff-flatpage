  49.     /**
  50.      * Method called to display a page content
  51.      *
  52.      * Read the requested slug from $_GET variables, try to get the associated page contents from database.<br />
  53.      * Display page contents if found and not forbidden.<br />
  54.      * Display a 403 template if contents are forbidden to requesting user.<br />
  55.      * Display a 404 template if contents are not found.

  60.     public function view() {
  61.     
  62.         $slug = cleanInput("get", "id", "string");
  63.  
  64.         $fp = flatpage::getFromSlug($this->_registry, $slug);
  65.  
  66.         // 404 not found
  67.         if(is_null($fp)) return $this->page404();
  68.         // 403 forbidden
  69.         if($fp->groups && !access::hasGroup($this->_registry, explode(",", $fp->groups))) return $this->page403();
  70.  
  71.         $this->_fp = $fp;
  72.         $text = preg_replace_callback("#{{(image[12]|video[12])\s*(width=(\d+))?\s*(height=(\d+))?\s*(position=(\w+))?\s*(lightbox=(true|false))?}}#", array($this, 'parseDescription'), $fp->text);
  73.         $text .= "<script>";
  74.         $text .= "(function() {
  75.             $$('.lightbox').cerabox({
  76.                 group: true,
  77.                 titleFormat: '{number} / {total} - {title}'
  78.             });
  79.         })()";
  80.  
  81.         $text .= "</script>";
  82.  
  83.         $this->_view->setTpl('flatpage_view');
  84.         $this->_view->assign('date', $this->_registry->dtime->view($fp->date, "date"));
  85.         $this->_view->assign('title', htmlVar($fp->title));
  86.         $this->_view->assign('subtitle', htmlVar($fp->subtitle));
  87.         $this->_view->assign('abstract', htmlVar($fp->abstract));
  88.         $this->_view->assign('text', htmlVar($text));
  89.  
  90.         return $this->_view->render();
  91.  
  92.     }

  148.     /**
  149.      * Page text parser
  150.      *
  151.      * Parse page text and replace custom images and video tags with html images and youtube iframes.
  152.      * 
  153.      * @param array[int]string $matches 
  154.      * @access private
  155.      * @return string 
  156.      */
  157.     private function parseDescription($matches) {
  158.  
  159.         if($matches[1]=='image1' || $matches[1]=='image2') {
  160.             $width = (isset($matches[3]) && $matches[3]) ? "width=\"".$matches[3]."px\"" : "";
  161.             $height = (isset($matches[5]) && $matches[5]) ? "height=\"".$matches[5]."px\"" : "";
  162.             $class = (isset($matches[7]) && $matches[7]) ? "class=\"".$matches[7]."\"" : "";
  163.             $lightbox = (isset($matches[9]) && $matches[9]=='true') ? true : false;
  164.  
  165.             $imgfile = $this->_fp->{$matches[1]};
  166.  
  167.             $img_tag = "<img src=\"".REL_UPLOAD."/flatpage/".$imgfile."\" alt=\"".$imgfile."\" $width $height $class />";
  168.  
  169.             if($lightbox) 
  170.                 return "<a href=\"".REL_UPLOAD."/flatpage/".$imgfile."\" class=\"lightbox\" />$img_tag</a>";
  171.             else return $img_tag;
  172.         }
  173.         elseif($matches[1]=='video1' || $matches[1]=='video2') {
  174.  
  175.             $width = isset($matches[3]) ? $matches[3] : 480;
  176.             $height = isset($matches[5]) ? $matches[5] : 390;
  177.             $class = isset($matches[5]) ? "class=\"".$matches[5]."\"" : "";
  178.  
  179.             $code = $this->_fp->{$matches[1]};
  180.  
  181.             return "<iframe width=\"$width\" height=\"$height\" src=\"http://www.youtube.com/embed/".$code."\" frameborder=\"0\" allowfullscreen></iframe>";
  182.  
  183.         }
  184.     }
